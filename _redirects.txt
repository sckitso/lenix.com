# Netlify rewrites.
# 200 = rewrite (URL stays clean in the browser bar).
# 301 = redirect (browser sees the new URL).

# Existing rule

#   https://lenixapp.com/referral/USER_abc123
# becomes internally:
#   /referral.html?ref=USER_abc123

/referral        /referral.html           200
